## Change 1: Inline Cancel / No-Show Confirmations in the Popover

### Context
The `AppointmentPopover` function (line ~973 in `appointments/page.tsx`) already renders status action buttons using `onStatusChange`. Currently Cancel and No Show fire immediately via `handleStatusChange`. Update the popover to show an inline confirmation step instead.

### Add local state to `AppointmentPopover`:
```ts
const [confirmingStatus, setConfirmingStatus] = useState<"CANCELLED" | "NO_SHOW" | null>(null);
const [cancelReason, setCancelReason] = useState("");
```

### New rendering logic for the status actions section:

Replace the current status actions `div` with this conditional:

**When `confirmingStatus === "CANCELLED"`** — show inline within the popover body:
```
┌─ "Cancel this appointment?" ──────────────────────┐
│  [reason input: "Reason (optional)"]              │
│  [Yes, Cancel — red]    [Go Back — gray]          │
└───────────────────────────────────────────────────┘
```
- "Yes, Cancel" calls `onStatusChange(appt.id, "CANCELLED", cancelReason)` then resets state
- "Go Back" sets `confirmingStatus` back to null

**When `confirmingStatus === "NO_SHOW"`** — show inline:
```
┌─ "Mark as no-show?" ──────────────────────────────┐
│  [Yes, No-Show — amber]    [Go Back — gray]       │
└───────────────────────────────────────────────────┘
```

**Otherwise** — show the existing action buttons, but change the Cancel and No Show buttons to set `confirmingStatus` instead of calling `onStatusChange` directly.

### Update `onStatusChange` prop and `handleStatusChange`:

The `AppointmentPopover`'s `onStatusChange` prop signature needs an optional reason parameter:
```ts
onStatusChange: (id: string, status: string, reason?: string) => void;
```

Update `handleStatusChange` in `AppointmentsPage` to accept and forward the optional reason:
```ts
const handleStatusChange = useCallback(async (appointmentId: string, newStatus: string, reason?: string) => {
  try {
    const res = await fetch(`/api/appointments/${appointmentId}/status`, {
      method: "PATCH",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ status: newStatus, ...(reason ? { reason } : {}) }),
    });
    if (res.ok) {
      queryClient.invalidateQueries({ queryKey: ["/api/appointments"] });
      queryClient.invalidateQueries({ queryKey: ["/api/appointments/agenda"] });
      if (popover && popover.appt.id === appointmentId) {
        setPopover(null);
      }
    }
  } catch {}
}, [queryClient, popover]);
```

### Inline confirmation styling:
- Container: `background: var(--bg-secondary)`, `border-radius: 8px`, `padding: 12px`, `margin-top: 8px`
- Title: `font-size: 13px`, `font-weight: 600`, `color: var(--text-primary)`, `margin-bottom: 8px`
- Reason input (Cancel only): full-width, standard input style, placeholder "Reason (optional)", `margin-bottom: 8px`

## Change 3: Drag-and-Drop Rescheduling

### Overview
Wrap the calendar grids in `DndContext`. Each patient appointment block becomes draggable. Each 15-minute time slot in both DayView and WeekView becomes a drop target. On drop, show a confirmation modal. On confirm, call PATCH. Internal blocks are never draggable.

### Setup in `AppointmentsPage`

Add state at the top of `AppointmentsPage`:
```ts

  appointmentId: string;
  patientName: string;
  typeName: string;
  fromTime: Date;
  toTime: Date;
  newProviderId?: string;
  durationMins: number;
} | null>(null);
const [activeAppt, setActiveAppt] = useState<AppointmentData | null>(null);
```

Add import at top of file:
```ts
import { DndContext, DragEndEvent, DragStartEvent, DragOverlay, useDroppable, useDraggable } from "@dnd-kit/core";
import { CSS } from "@dnd-kit/utilities";
```

Wrap the entire calendar area (the `div` containing `{view === "day" && ...}` etc.) in:
```tsx
<DndContext
  onDragStart={(event: DragStartEvent) => {
    const appt = filteredAppointments.find(a => a.id === event.active.id);
    setActiveAppt(appt || null);
  }}
  onDragEnd={(event: DragEndEvent) => {
    setActiveAppt(null);
    if (!event.over || !activeAppt) return;

    const overId = String(event.over.id);
    // Drop target IDs are encoded as: "slot-[isoDate]-[hour]-[minute]-[providerId|'same']"
    if (!overId.startsWith("slot-")) return;

    const parts = overId.split("-");
    // parts: ["slot", isoDate, hour, minute, providerId]
    const isoDate = parts[1];
    const hour = parseInt(parts[2]);
    const minute = parseInt(parts[3]);
    const newProviderId = parts[4] === "same" ? activeAppt.provider?.id : parts[4];

    const originalStart = new Date(activeAppt.startTime);
    const originalEnd = new Date(activeAppt.endTime);
    const durationMins = Math.round((originalEnd.getTime() - originalStart.getTime()) / 60000);

    const newStart = new Date(isoDate);
    newStart.setHours(hour, minute, 0, 0);

    // No-op if same time and same provider
    if (
      newStart.getTime() === originalStart.getTime() &&
      newProviderId === activeAppt.provider?.id
    ) return;

    const patientName = activeAppt.patient
      ? `${activeAppt.patient.firstName} ${activeAppt.patient.lastName}`
      : activeAppt.title;
    const typeName = activeAppt.configuredType?.name || activeAppt.appointmentCategory;

    setRescheduleConfirm({
      appointmentId: activeAppt.id,
      patientName,
      typeName,
      fromTime: originalStart,
      toTime: newStart,
      newProviderId,
      durationMins,
    });
  }}
>
  {/* ... calendar views ... */}
  <DragOverlay>
    {activeAppt && (
      <div style={{
        width: "160px",
        backgroundColor: colorWithAlpha(getApptColor(activeAppt), 0.9),
        borderLeft: `3px solid ${getApptColor(activeAppt)}`,
        borderRadius: "6px",
        padding: "4px 8px",
        fontSize: "12px",
        fontWeight: 600,
        color: "#1e293b",
        boxShadow: "0 8px 24px rgba(0,0,0,0.2)",
        opacity: 0.9,
        cursor: "grabbing",
      }}>
        {activeAppt.patient
          ? `${activeAppt.patient.firstName} ${activeAppt.patient.lastName}`
          : activeAppt.title}
      </div>
    )}
  </DragOverlay>
</DndContext>
```

```ts
if (body.isInternal && body.recurrence) {
  const { frequency, customDays, endType, endAfter, endDate } = body.recurrence;

  // Generate target dates
  const dates: Date[] = [];
  const baseStart = new Date(body.startTime);
  const durationMs = new Date(body.endTime).getTime() - baseStart.getTime();
  let current = new Date(baseStart);
  const maxOccurrences = Math.min(endType === "after" ? endAfter : 52, 52);
  const endDateLimit = endType === "date" && endDate ? new Date(endDate) : null;

  while (dates.length < maxOccurrences) {
    if (endDateLimit && current > endDateLimit) break;
    const dayOfWeek = current.getDay();
    const include =
      frequency === "DAILY" ||
      (frequency === "WEEKLY" && current.getTime() !== baseStart.getTime() ? dayOfWeek === baseStart.getDay() : true) ||
      (frequency === "CUSTOM" && customDays.includes(dayOfWeek));
    if (include || dates.length === 0) {
      dates.push(new Date(current));
    }
    current.setDate(current.getDate() + 1);
    if (dates.length >= maxOccurrences) break;
  }

  // Create all blocks in a transaction
  const appointments = await prisma.$transaction(
    dates.map(d => {
      const s = new Date(d);
      const e = new Date(s.getTime() + durationMs);
      return prisma.appointment.create({
        data: {
          practiceId: practice.id,
          providerId: body.providerId,
          createdById: user.id,
          blockTypeId: body.blockTypeId || null,
          title: body.title || "Internal Block",
          isInternal: true,
          startTime: s,
          endTime: e,
          bufferMins: body.bufferMins || 0,
          appointmentCategory: "OTHER",
          status: "CONFIRMED",
          notes: body.notes || null,
          roomName: body.roomName || null,
        },
      });
    })
  );

  return NextResponse.json({ appointments, count: appointments.length }, { status: 201 });
}
```

For non-recurring internal blocks (and all patient appointments), the existing single-create path continues unchanged — just add `blockTypeId: body.blockTypeId || null` to the `data` object.

### New API route: `GET /api/appointments/[id]/history`

Create `app/api/appointments/[id]/history/route.ts`:
```ts
import { NextRequest, NextResponse } from "next/server";
import { prisma } from "@/lib/db";
import { requireUser } from "@/lib/auth";

export async function GET(
  req: NextRequest,
  { params }: { params: Promise<{ id: string }> }
) {
  try {
    const { practice } = await requireUser();
    const { id } = await params;

    // Verify the appointment belongs to this practice
    const appointment = await prisma.appointment.findFirst({
      where: { id, practiceId: practice.id, deletedAt: null },
    });
    if (!appointment) {
      return NextResponse.json({ error: "Appointment not found" }, { status: 404 });
    }

    const activities = await prisma.activity.findMany({
      where: {
        practiceId: practice.id,
        metadata: {
          path: ["appointmentId"],
          equals: id,
        },
      },
      orderBy: { createdAt: "desc" },
      include: {
        user: { select: { id: true, firstName: true, lastName: true } },
      },
    });

    return NextResponse.json(activities);
  } catch (error: any) {
    if (error.message === "Unauthorized") {
      return NextResponse.json({ error: "Unauthorized" }, { status: 401 });
    }
    return NextResponse.json({ error: "Internal server error" }, { status: 500 });
  }
}
```
